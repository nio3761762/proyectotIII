const agregarItemAVenta = () => {
  if (!selectedItem.value || cantidadItem.value <= 0) {
    showToastMessage('Selecciona un ítem y una cantidad válida');
    return;
  }

  const item = selectedItem.value;
  let precio = 0;

  if (item.IdPromocion) {
    for (const precioventa of item.Promocionproducto) {
      const precioProducto = precioventa.Producto.Precio[0]?.Precio || 0;
      const descuento = precioventa.Descuento / 100;
      precio += precioventa.Cantidad * (precioProducto * (1 - descuento));
    }
  } else {
    precio = Array.isArray(item.Precio) ? item.Precio[0]?.Precio || 0 : item.PrecioVenta || 0;
  }

  const nuevoItem = {
    id: item.IdProducto || item.IdPromocion || item.IdPaquete,
    type: selectedItemType.value,
    nombre: item.Nombre,
    Cantidad: cantidadItem.value,
    precioUnitario: precio,
    subtotal: cantidadItem.value * precio,
  };

  switch (selectedItemType.value) {
    case 'producto':
      venta.value.ventas.detalles.Producto.push(nuevoItem);
      break;
    case 'promocion':
      venta.value.ventas.detalles.Promocion.push(nuevoItem);
      break;
    case 'paquete':
      venta.value.ventas.detalles.Paquete.push(nuevoItem);
      break;
  }

  deseleccionarItem();
  cantidadItem.value = 1;
  showToastMessage('Ítem agregado a la venta');
};

  let index = Producto.findIndex(item => item.id === itemToRemove.id);
  if (index !== -1) {
    Producto.splice(index, 1);
  } else {
    index = Promocion.findIndex(item => item.id === itemToRemove.id);
    if (index !== -1) {
      Promocion.splice(index, 1);
    } else {
      index = Paquete.findIndex(item => item.id === itemToRemove.id);
      if (index !== -1) {
        Paquete.splice(index, 1);
      }
    }
  }
  showToastMessage('Ítem eliminado de la venta');
