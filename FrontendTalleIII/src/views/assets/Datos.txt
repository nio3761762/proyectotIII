const openPhotoModal = async (datos) =>{

  console.log( previewUrl.value  )
  isPhotoModalOpen.value = true;
  try {
    const response = await fetch(previewUrl.value);
    const blob = await response.blob();

    // Obtener la extensión desde el tipo MIME (ej. image/jpeg → jpeg)
    const mimeType = blob.type; // ejemplo: "image/jpeg"
    const extension = mimeType.split('/')[1] || 'jpg';

    // Nombre genérico con extensión adecuada
    const fileName = `imagen.${extension}`;

    // Crear el archivo
    archivo.value = new File([blob], fileName, { type: mimeType });

    isPhotoModalOpen.value = true;
  } catch (error) {
    console.error('Error al convertir la URL en archivo:', error);
  }
  console.log(archivo.value)
};
